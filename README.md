# Приложение создано в рамках ТЗ


# Архитектура
я использовал слоистую архитектуру Clean подразделяемую на уровне
Data -данные
    di

Domain - бизнес логика
    services 

Screen - интерфейс
    screen - экраны
    resurses - ресурсы
       -color.dart - цвета
       -image.dart - картинки


# Структура

lib/
│
├── main.dart                  # Точка входа в приложение
├── route.dart                 # Все маршруты приложения
│
├── data/                      # Работа с данными и DI
│   └── di/
│       ├── get_it_servis.dart # Инициализация GetIt
│       └── locator.dart       # Регистрация зависимостей
│
├── domain/                    # Бизнес-логика и сервисы
│   └── services/
│       ├── navigator_service.dart        # Навигация через сервис
│       └── shared_preferens_service.dart # Работа с SharedPreferences
│
├── ui/                        # UI слой приложения
│   ├── resurses/              # Ресурсы приложения
│   │   ├── colors.dart        # Цвета
│   │   └── image.dart         # Пути к изображениям
│   │
│   └── screens/               # Все экраны приложения
│       ├── splash_screen/     # Splash Screen
│       ├── onboarding_screen/ # Онбординг
│       ├── paywall_screen/    # Экран подписки
│       ├── home_screen/       # Главный экран
│       └── widgets/           # Общие UI компоненты
│
assets/
└── images/                    # Картинки приложения
